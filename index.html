document.getElementById('attendanceForm').addEventListener('submit', async function(event) { event.preventDefault(); // Prevent form from refreshing the page const name = document.getElementById('name').value; const status = document.getElementById('status').value; // Insert attendance data into Supabase try { const { data, error } = await supabase .from('attendance') .insert([{ name, status }]); if (error) { console.error('Error inserting data:', error); alert('Error saving record'); } else { console.log('Data saved successfully:', data); // Fetch and display updated records after insertion fetchRecords(); document.getElementById('attendanceForm').reset(); } } catch (error) { console.error('Error:', error); } }); // Fetch existing records from Supabase async function fetchRecords() { try { const { data, error } = await supabase .from('attendance') .select('*') // Select all records .order('created_at', { ascending: false }); // Order by creation time (if you have a timestamp column) if (error) { console.error('Error fetching data:', error); alert('Error fetching records'); } else { console.log('Fetched records:', data); const tableBody = document.getElementById('attendanceTableBody'); tableBody.innerHTML = ''; // Clear existing rows before adding new ones data.forEach(record => { const row = document.createElement('tr'); const nameCell = document.createElement('td'); const statusCell = document.createElement('td'); nameCell.textContent = record.name; statusCell.textContent = record.status; row.appendChild(nameCell); row.appendChild(statusCell); tableBody.appendChild(row); }); } } catch (error) { console.error('Error fetching records:', error); } } // Call the function when the page loads fetchRecords();
